import javafx.application.Application;
import javafx.scene.Scene;
import javafx.scene.control.*;
import javafx.scene.layout.VBox;
import javafx.stage.Stage;

public class GitHubAnalyzerUI extends Application {

    @Override
    public void start(Stage stage) {
        TextField usernameField = new TextField();
        usernameField.setStyle("-fx-font-size: 15px;");
        usernameField.setPromptText("Enter  Username As For GitHub!");

        Button analyzeBtn = new Button("Analyze");

        Label resultLabel = new Label();

        analyzeBtn.setOnAction(e -> {
            String username = usernameField.getText();

            String result = GitHubService.analyzeUser(username);

            resultLabel.setText(result);
        });

        VBox root = new VBox(1, usernameField, analyzeBtn, resultLabel);

        Scene scene = new Scene(root, 400, 400);
        stage.setTitle("GitHub Credibility Checker");
        stage.setScene(scene);
        stage.show();
    }

    public static void main(String[] args)
    {

        launch();
    }
}
